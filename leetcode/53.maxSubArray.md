class Solution {
    public int maxSubArray(int[] nums) {
        int[] dp = new int[nums.length];//dp[i]定义为数组nums 中已num[i] 结尾的最大连续子串和
        dp[0] = nums[0];
        int max = dp[0];
        for(int i = 1; i < nums.length; i++) {
            dp[i] = Math.max(dp[i - 1] + nums[i], nums[i]);
            if(max < dp[i]) {
                max = dp[i];
            }
        }
        return max;
    }
}
