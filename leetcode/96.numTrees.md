class Solution {
    public int numTrees(int n) {
        //dp[i]表示1到i个节点组成的二叉搜索树的个数
        int[] dp = new int[n + 1];
        dp[0] = 1;
        for(int i = 1; i < n + 1; i++) {
            for(int j = 1; j <= i; j++) {
                dp[i] += dp[j - 1] * dp[i - j];
            }
        }
        return dp[n];
    }
}
