

```java
class Solution {
    public int climbStairs(int n) {
        if(n < 2) return n;
        int[] dp = new int[n + 1];
        dp[1] = 1;
        dp[2] = 2;
        for(int i = 3; i < n + 1; i++) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        return dp[n];
    }
}
```

```java
class Solution {
    public int climbStairs(int n) {
        if(n < 3) return n;
        int a = 1;
        int b = 2;
        int c = 0;
        for(int i = 3; i < n + 1; i++) {
            c = a + b;
            a = b;
            b = c;
        }
        return c;
    }
}
```
