class Solution {
    public boolean containsNearbyDuplicate(int[] nums, int k) {
        //维护一个大小为k的滑动窗口
        Set<Integer> set = new HashSet<>();
        for(int i = 0; i < nums.length; i++) {
            if(set.contains(nums[i])) {
                return true;
            }
            set.add(nums[i]);
            if(set.size() > k) {
                set.remove(nums[i - k]);
            }
        }
        return false;
    }
}
