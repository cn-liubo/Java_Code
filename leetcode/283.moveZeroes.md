class Solution {
    public void moveZeroes(int[] nums) {
        int l = 0, r = 0;
        int n = nums.length;
        while(r < n) {
            if(nums[r] != 0) {
                swap(nums, l, r);
                l++;
            }
            r++;
        }
    }

    private void swap(int[] nums, int i, int j) {
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
    }
}
