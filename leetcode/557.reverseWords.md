class Solution {
    public String reverseWords(String s) {
        String[] str = s.split(" ");
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < str.length; i++) {
            char[] chars = str[i].toCharArray();
            int l = 0;
            int r = chars.length - 1;
            while(l <= r) {
                char tmp = chars[l];
                chars[l] = chars[r];
                chars[r] = tmp;
                l++;
                r--;
            }
            sb.append(chars);
            sb.append(" ");
        }
        return sb.toString().trim();
    }
}
