```java
class Solution {
    public int tribonacci(int n) {
        if(n < 2) return n;
        int[] dp = new int[n + 1];
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 1;
        for(int i = 3; i < n + 1; i++) {
            dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3];
        }
        return dp[n];
    }
}
```

```java
class Solution {
    public int tribonacci(int n) {
        if(n < 3) return n == 0 ? 0 : 1;
        int a = 0;
        int b = 1;
        int c = 1;
        int d = 0;
        for(int i = 3; i < n + 1; i++) {
            d = a + b + c;
            a = b;
            b = c;
            c = d;
        }
        return d;
    }
}
```
