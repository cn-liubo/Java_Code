/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public TreeNode buildTree(int[] preorder, int[] inorder) {
        int n = inorder.length;
        //构建哈希索引
        Map<Integer, Integer> inMap = new HashMap<>();
        for(int i = 0; i < n; i++) {
            inMap.put(inorder[i], i);
        }
        return build(preorder, 0, n - 1, inorder, 0, n - 1, inMap);
    }
    public TreeNode build(int[] preorder, int pre_left, int pre_right, int[] inorder, int in_left, int in_right, Map<Integer, Integer> inMap) {
        if(pre_left > pre_right || in_left > in_right) {
            return null;
        }

        //前序遍历的第一个结点就是根节点
        int pre_root = pre_left;
        //在中序遍历中定位出根结点
        int in_root = inMap.get(preorder[pre_root]);
        TreeNode root = new TreeNode(preorder[pre_root]);
        //左子树中的结点数目
        int left_size = in_root - in_left;
        root.left = build(preorder, pre_left + 1, pre_left + left_size, inorder, in_left, in_root - 1, inMap);
        root.right = build(preorder, pre_left + left_size + 1, pre_right, inorder, in_root + 1, in_right, inMap);
        return root;
    }
}
