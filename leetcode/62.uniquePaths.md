解法一：深度优先遍历

```java
class Solution {
    public int uniquePaths(int m, int n) {
        return dfs(1, 1, m, n);
    }

    private int dfs(int i, int j, int m, int n) {
        if(i > m || j > n) return 0;
        if(i == m && j == n) return 1;
        return dfs(i + 1, j, m, n) + dfs(i, j + 1, m, n);
    }
}
```

解法二：动态规划

```java
class Solution {
    public int uniquePaths(int m, int n) {
        int[][] dp = new int[m][n];
        for(int i = 0; i < m; i++) dp[i][0] = 1;
        for(int i = 0; i < n; i++) dp[0][i] = 1;
        for(int i = 1; i < m; i++) {
            for(int j = 1; j < n; j++) {
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
        return dp[m - 1][n - 1];
    }
}
```
