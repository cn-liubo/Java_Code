class Solution {
    public boolean isValid(String s) {
        Deque<Character> stack = new LinkedList<>();
        for(char c : s.toCharArray()) {
            if(c == '(' || c == '[' || c == '{') {
                stack.push(c);
            } else {
                char t = stack.pop();
                if((c == ')' && t != '(') || (c == ']' && t != '[') || (c == '}' && t != '{')) {
                    return false;
                }
            }
        }
        return stack.isEmpty();
    }
}
